num = 0
while num <= 10: # num > 10 (종료 조건) (탈출 조건)
    print("반복")
    num += 1     # 10번 반복 ==> num += 1
print("종료")

num = 0
while num <= 10: # 원하는 숫자만큼 반복을 시킨다
    print("반복")
    num += 1
#=================================================================

while end == 0: # 계속 반복시키다가 원하는 떄 종료시킨다  # end != 0 ==> (종료 조건)
    print("반복1")
    end = int(input("계쏙하고 싶으면 0을 입력하세요 >>> "))
print("종료")
