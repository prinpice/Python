```python
# 문자열 인덱싱
# 앞에서는 0부터 시작하고
# 뒤에서부터는 -1 부터 시작한다.

str1 = "헬로우월드"

print(str1)  # 헬로우월드

print(str1[0]) # 헬
print(str1[-1]) # 드

# 문제 1) "우" 를 2가지 방법으로 출력해보자
print(str1[2])
print(str1[-3])

# 문제 2 ) 주민번호를 입력받고 남자인지 여자인지 출력해보자 880320-1829917
str = input("주민번호를 입력하세요 >>> ")
if str[7] == "1" or str[7] == "3" :
    print("남자입니다")
elif str[7] =="2" or str[7] == "4":
    print("여자입니다")
else:
    print("잘못 입력하셨습니다")
```
