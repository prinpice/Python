```python
# 숫자 출력

print("10") #10
print(10)  #10

# +  ==> 더하기
print("10+10")  #10+10
print(10+10)  #20


# 자료형  ==> 크게 문자 와 숫자가 있다.
# 더하기 빼기 곱하기 등등을 연산이라고 하는데 숫자와 문자는 연산결과가 다르다.

# 문자  ++> ""(쌍따옴표) 안에 있는 내용은 문자이다.

# 10 + 10 = 20
# 문자와 숫자를 같이 출력
print("10 + 10 = ", end="")
print(20)
print(10, "+", 10, "=", 10+10)  # 10 + 10 = 20  문자와 숫자를 섞어서 사용

# 문제 1) -(빼기) *(곱하기) /(나누기) 를 사용하여 위에처럼 출력해보세요
print(10, "-", 10, "=", 10-10) # 10 - 10 = 0
print(10, "*", 10, "=", 10*10) # 10 * 10 = 100
print(10, "/", 10, "=", 10/10) # 10 / 10 = 1






```
