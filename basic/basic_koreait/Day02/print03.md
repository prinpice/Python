```python
# 문제 1) 10 * 3 = 30 을 출력해보세요

print(10, "*", 3, "=", 10*3) # 10 * 3 = 30

# 1. 저장  ==> 1. 변수 2. 리스트(배열) 3. 클래스
# 2. 찾기  ==> 1. 연산자(4종류) 2. 조건문(if) 3. 반복문(while)

# 연산자 4종류 ==> 1. 산술연산자 ( + - * /)
#  / ==> 1. // (몫), 2. / (나누기), 3. % (나머지)
print(5//3) # 1    ==> 몫
print(5/3) # 1.66666666667    ==> 나누기
print(5%3) # 2    ==> 나머지






```
