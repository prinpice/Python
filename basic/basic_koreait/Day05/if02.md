# if02

```python
# if 를 여러개 쓰는 것과 elif 를 쓰는 것과의 차이
num = 100
if num > 90:
    print("a")
if num > 80:
    print("b")
if num > 70:
    print("c")
else:
    print("f")   # a b c

num = 100
if num > 90:
    print("a")
elif num > 80:
    print("b")
elif num > 70:
    print("c")
else:
    print("f")   # a

# 숫자 하나를 입력받고 짝수 홀수

```
